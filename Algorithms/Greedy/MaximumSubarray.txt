Given an integer array nums, find the subarray with the largest sum, and return its sum.

class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        sum=nums[0]
        maxSum=sum
        for i in range(1,len(nums)):
            sum=max(sum,0)
            sum+=nums[i]
            maxSum=max(maxSum,sum)
        return maxSum
